CREATE TRIGGER islamicacademy_product_to_main_product_trigger_insert
AFTER INSERT ON products_islamicacademy_product
FOR EACH ROW
BEGIN
    INSERT INTO products_main_product(
        product_name,
        product_slug,
        created_at,
        modified_at,
        product_type,
        status,
        description,
        amount_usd,
        stock_status,
        source
    ) VALUES (
        NEW.name,
        NEW.product_slug,
        NEW.created_at,
        NEW.modified_at,
        NEW.type,
        NEW.status,
        NEW.description,
        NEW.price,
        NEW.stock_status,
        'Islamic Academy'
    );
END;
